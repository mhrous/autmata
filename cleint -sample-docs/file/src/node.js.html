    /**
 * Node class
 */
export default class Node {
  /**
   *
   * @param {String} label
   * @param {Boolean} start
   * @param {Boolean} end
   * @param {String} massege
   */
  constructor(label, start = false, end = false, massege = "") {
    /** @type {string} */
    this.label = label;

    /** @type {string} */
    this.id = label;

    /** @type {Map Object Between nodes and transitions} يربط بين الانتقال والعقدة الهدف */
    this.edges = {};

    /** @type {Boolean} */
    this.start = start;
    /** @type {Boolean} */
    this.end = end;
    /** @type {String} */
    this.massege = massege;
    this.level = "";
  }
  /**
   *
   * @param {The character expresses the transition } state
   * @param {target Node} to

   */
  addEdge(state, to) {
    this.edges[state] = to;
  }

  /**
   *
   * @param {The character expresses the transition} state
   * @return {target Node}

   */
  getPossition(state) {
    return this.edges[state];
  }
}
